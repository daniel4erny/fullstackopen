    server0 --> client2(["browser sends GET on /notes"])
    client2 --> server1["server sends index.html"]

    server1 --> client3["Browser sees main.css and main.js in sources and sends GET on the file paths"]

    server2 --> client4["browser executes .js code that fetches data.json"]

    client4 --> server3["server sends the .json file, now including the new note"]

    server3 --> client5["browser calls rendering function, new note appears on the page"]
